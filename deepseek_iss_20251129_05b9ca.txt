begin

  // Result := CheckPythonInstallation;
  
  // Verificar requisitos do sistema
  Result := CheckSystemRequirements;
end;

function CheckSystemRequirements(): Boolean;
begin
  // Verificar versão do Windows
  if GetWindowsVersion < $06010000 then // Windows 7 ou superior
  begin
    MsgBox('Este aplicativo requer Windows 7 ou superior.', mbError, MB_OK);
    Result := False;
  end
  else
    Result := True;
end;
